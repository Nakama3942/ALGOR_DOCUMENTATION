45memcell_t ALGOR::getMemoryCell(memcell_t right_adjust, memcell_t left_adjust)
46{
47 memcell_t *cells = new memcell_t[10];
48 memcell_t cell = cells[0];
49 for (ubit32_t i = 1; i < 8; i++)
50 {
51 cell >>= (memcell_t)cells[i];
52 cell <<= (memcell_t)cells[i + 1];
53 cell ^= (memcell_t)cells[i + 2];
54 if (right_adjust != 0)
55 {
56 cell >>= right_adjust;
57 }
58 if (left_adjust != 0)
59 {
60 cell <<= left_adjust;
61 }
62 }
63 delete[] cells;
64 return cell;
65}

67template <typename type_array>
68Array<type_array> *ALGOR::create_struct(const asize_t &SIZE,
69 bool mem_allocation)
70{
71 if (SIZE == 0)
72 {
73 throw EXCEPTION_SET::void_data();
74 }
75 Array<type_array> *ARRAY = new Array<type_array>;
76 ARRAY->array_size = SIZE;
77 if (mem_allocation)
78 {
79 ARRAY->array = new type_array[SIZE];
80 }
81 return ARRAY;
82}

84template <typename type_array>
85void ALGOR::generate_struct(Array<type_array> *&Array,
86 const sbit64_t &min_limit,
87 const sbit64_t &max_limit,
88 const ubit64_t seed,
89 const ubit32_t denominator)
90{
91 if (denominator == 0)
92 {
93 throw EXCEPTION_SET::division_by_zero("The Denominator variable is designed to convert the generated integer into a"
94 "fractional number or find its part. It's a divisor, so it can't be zero!");
95 }
96 RANDOM::LCM RanGen(seed);
97 for (asize_t i = 0; i < Array->array_size; i++)
98 {
99 Array->array[i] = (min_limit + (RanGen.rand() % (max_limit - min_limit))) / (fbit32_t)denominator;
100 }
101}

103template <typename type_array>
104void ALGOR::remove_struct(Array<type_array> *&Array)
105{
106 if (Array == nullptr || Array->array == nullptr)
107 {
108 throw EXCEPTION_SET::void_data();
109 }
110 delete[] Array->array;
111 delete (Array);
112 Array = nullptr;
113}
