template <typename type_array>
class printer : public Printer<type_array>
{
public:
void setData(Array<type_array> **ARRAY)
{
Array = ARRAY;
}
void print() override
{
for (asize_t i = 0; i < (*Array)->array_size; i++)
{
cout << " " << (*Array)->array[i];
}
cout << "\n";
}
private:
Array<type_array> **Array;
};
