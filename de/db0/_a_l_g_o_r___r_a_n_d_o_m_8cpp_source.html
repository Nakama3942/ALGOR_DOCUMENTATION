41void ALGOR::RANDOM::RC4::crypto_srand(const byte1_t *key, byte4_t ksize)
42{
43 ubit8_t j = 0;
44 for (byte4_t i = 0; i < 256; i++)
45 {
46 Sbox[i] = i;
47 }
48 for (byte4_t i = 0; i < 256; i++)
49 {
50 j = j + Sbox[i] + (ubit8_t)key[i % ksize];
51 swap(Sbox[i], Sbox[j]);
52 }
53}

55void ALGOR::RANDOM::RC4::crypto_rand(byte1_t *output, byte4_t size)
56{
57 ubit8_t i = 0, j = 0, t;
58 for (byte4_t k = 0; i < size; k++)
59 {
60 i += 1;
61 j += Sbox[i];
62 swap(Sbox[i], Sbox[j]);
63 t = Sbox[i] + Sbox[j];
64 output[k] = (ubit32_t)Sbox[t];
65 }
66}

88byte4_t ALGOR::RANDOM::MersenneTwister::IRandom(byte4_t min, byte4_t max)
89{
90 if (max <= min)
91 {
92 return max == min ? min : 0x80000000;
93 }
94 byte4_t rand_int = byte4_t((fbit64_t)(ubit32_t)(max - min + 1) * Random() + min);
95 if (rand_int > max)
96 {
97 rand_int = max;
98 }
99 return rand_int;
100}

102byte4_t ALGOR::RANDOM::MersenneTwister::IRandomX(byte4_t min, byte4_t max)
103{
104 if (max <= min)
105 {
106 return max == min ? min : 0x80000000;
107 }
108 ubit32_t len_interval; // Length of interval
109 ubit64_t long_rbi; // Random bits * interval
110 ubit32_t iran; // long_rbi / 2^32
111 ubit32_t remainder; // long_rbi % 2^32
112
113 len_interval = ubit32_t(max - min + 1);
114 if (len_interval != LastInterval)
115 {
116 RejectionLimit = ubit32_t(((ubit64_t)1 << 32) / len_interval) * len_interval - 1;
117 LastInterval = len_interval;
118 }
119 do
120 {
121 long_rbi = (ubit64_t)BRandom() * len_interval;
122 iran = (ubit32_t)(long_rbi >> 32);
123 remainder = (ubit32_t)long_rbi;
124 } while (remainder > RejectionLimit);
125 return (sbit32_t)iran + min;
126}

133ubit32_t ALGOR::RANDOM::MersenneTwister::BRandom()
134{
135 ubit32_t resulting_bit;
136 if (mersenne_twister_index >= 624)
137 {
138 const ubit32_t LOWER_MASK = (1LU << 31) - 1; // Lower bits
139 const ubit32_t UPPER_MASK = 0xFFFFFFFF << 31; // Upper bits
140 static const ubit32_t mask[2] = {0, 0x9908B0DF};
141 byte4_t counter;
142 for (counter = 0; counter < 624 - 397; counter++)
143 {
144 resulting_bit = (mersenne_twister[counter] & UPPER_MASK) | (mersenne_twister[counter + 1] & LOWER_MASK);
145 mersenne_twister[counter] = mersenne_twister[counter + 397] ^ (resulting_bit >> 1) ^ mask[resulting_bit & 1];
146 }
147 for (; counter < 624 - 1; counter++)
148 {
149 resulting_bit = (mersenne_twister[counter] & UPPER_MASK) | (mersenne_twister[counter + 1] & LOWER_MASK);
150 mersenne_twister[counter] = mersenne_twister[counter + (397 - 624)] ^ (resulting_bit >> 1) ^ mask[resulting_bit & 1];
151 }
152 resulting_bit = (mersenne_twister[624 - 1] & UPPER_MASK) | (mersenne_twister[0] & LOWER_MASK);
153 mersenne_twister[624 - 1] = mersenne_twister[397 - 1] ^ (resulting_bit >> 1) ^ mask[resulting_bit & 1];
154 mersenne_twister_index = 0;
155 }
156 resulting_bit = mersenne_twister[mersenne_twister_index++];
157 resulting_bit ^= resulting_bit >> 11;
158 resulting_bit ^= (resulting_bit << 7) & 0x9D2C5680;
159 resulting_bit ^= (resulting_bit << 15) & 0xEFC60000;
160 resulting_bit ^= resulting_bit >> 18;
161 return resulting_bit;
162}

164void ALGOR::RANDOM::MersenneTwister::Init0(byte4_t seed)
165{
166 // Basic initialization procedure
167 const ubit32_t factor = 1812433253UL;
168 mersenne_twister[0] = seed;
169 for (mersenne_twister_index = 1; mersenne_twister_index < 624; mersenne_twister_index++)
170 {
171 mersenne_twister[mersenne_twister_index] = (
172 factor * (
173 mersenne_twister[mersenne_twister_index - 1] ^ (
174 mersenne_twister[mersenne_twister_index - 1] >> 30
175 )
176 ) + mersenne_twister_index
177 );
178 }
179}

182fbit64_t ALGOR::RANDOM::test_random_on_Pearsons_coefficient(ubit32_t left_limit, ubit32_t right_limit)
183{
184 //Створюю об'єкт генератора
185 memcell_t cell = getMemoryCell();
186 cell >>= 32;
187 Generator gen(cell);
188
189 //Генерую масив
190 fbit64_t *arr = new fbit64_t[1000000];
191 for (ubit32_t i = 0; i < 1000000; i++)
192 {
193 arr[i] = (left_limit + (gen.rand() % (right_limit - left_limit))) / 1000000.;
194 }
195
196 //Створюю 100 піддіапазонів
197 fbit64_t *bucket = new fbit64_t[100];
198 for (ubit32_t i = 0; i < 100; i++)
199 {
200 bucket[i] = 0;
201 }
202
203 //Розраховую 100 піддіапазонів
204 for (ubit32_t i = 0; i < 1000000; i++)
205 {
206 ubit32_t bucket_index = (ubit32_t)(arr[i] * 100);
207 bucket[bucket_index] += 1;
208 }
209
210 //Розраховую критерій узгодженості Пірсона та найбільші відхилення
211 //в меншу та більшу сторони; записую дані до LOG-файлу
212 fbit64_t Pj = 10000;
213 fbit64_t pX = 0;
214 for (ubit8_t i = 0; i < 100; i++)
215 {
216 pX += ((bucket[i] - Pj) * (bucket[i] - Pj)) / Pj;
217 }
218
219 //Звільняю пам'ять
220 delete[] bucket;
221 delete[] arr;
222
223 return pX;
224}
