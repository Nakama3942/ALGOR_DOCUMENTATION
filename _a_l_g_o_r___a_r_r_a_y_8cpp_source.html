25template <typename type_array>
26type_array ALGOR::ArrayProcessing<type_array>::minimum(const type_array *Array,
27 const asize_t &array_size)
28{
29 type_array point_min = Array[0];
30 for (asize_t i = 1; i < array_size; i++)
31 {
32 if (point_min > Array[i])
33 {
34 point_min = Array[i];
35 }
36 }
37 return point_min;
38}

55template <typename type_array>

57 const asize_t &array_size)
58{
59 for (asize_t i = 0; i < array_size - 1; i++)
60 {
61 if (Array[i] > Array[i + 1])

65 }

76template <typename type_array>
77type_array *ALGOR::ArrayProcessing<type_array>::lower_bound(type_array *first,
78 type_array *last,
79 const type_array &value)
80{
81 asize_t count = distance(first, last), step = 0, iterator = 0;
82 while (count > 0)
83 {
84 step = count / 2;
85 iterator = step;
86 if (first[iterator] < value)
87 {
88 first = &first[++iterator];
89 count -= (step + 1);
90 }
91 else
92 {
93 count = step;
94 }
95 }
96 return first;
97}

99template <typename type_array>
100type_array *ALGOR::ArrayProcessing<type_array>::upper_bound(type_array *first,
101 type_array *last,
102 const type_array &value)
103{
104 asize_t count = distance(first, last), step = 0, iterator = 0;
105 while (count > 0)
106 {
107 step = count / 2;
108 iterator = step;
109 if (!(value < first[iterator]))
110 {
111 first = &first[++iterator];
112 count -= (step + 1);
113 }
114 else
115 {
116 count = step;
117 }
118 }
119 return first;
120}

122template <typename type_array>
123void ALGOR::ArrayProcessing<type_array>::addElement(type_array *&Array,
124 asize_t &array_size,
125 const type_array &value,
126 const asize_t position)
127{
128 if (array_size == 0xffffffff)
129 {
130 throw EXCEPTION_SET::memory_overflow();
131 }
132 array_size++;
133 if (array_size == 1)
134 {
135 Array = new type_array[array_size]{value};
136 return;
137 }
138 type_array *temp_Array = new type_array[array_size];
139 for (asize_t i = 0; i < array_size; i++)
140 {
141 position > i ? temp_Array[i] = Array[i] : (position == i ? temp_Array[i] = value : temp_Array[i] = Array[i - 1]);
142 }
143 delete[] Array;
144 Array = temp_Array;
145}

147template <typename type_array>
148void ALGOR::ArrayProcessing<type_array>::subtractElement(type_array *&Array,
149 asize_t &array_size,
150 const asize_t position)
151{
152 if (array_size == 0)
153 {
154 throw EXCEPTION_SET::void_data();
155 }
156 if (position >= array_size)
157 {
158 throw EXCEPTION_SET::position_failure();
159 }
160 array_size--;
161 if (array_size == 0)
162 {
163 delete[] Array;
164 return;
165 }
166 type_array *temp_Array = new type_array[array_size];
167 for (asize_t i = 0; i < array_size; i++)
168 {
169 position > i ? temp_Array[i] = Array[i] : temp_Array[i] = Array[i + 1];
170 }
171 delete[] Array;
172 Array = temp_Array;
173}

175template <typename type_array>
176void ALGOR::ArrayProcessing<type_array>::subtractValue(type_array *&Array,
177 asize_t &array_size,
178 const type_array &value)
179{
180 if (array_size == 0)
181 {
182 throw EXCEPTION_SET::void_data();
183 }
184 ubit32_t deletions_count = 0;
185 type_array *temp_Array = new type_array[array_size];
186 for (asize_t i = 0; i < array_size; i++)
187 {
188 Array[i] != value ? temp_Array[i - deletions_count] = Array[i] : deletions_count++;
189 }
190 if (deletions_count == 0)
191 {
192 delete[] temp_Array;
193 throw EXCEPTION_SET::value_failure();
194 }
195 delete[] Array;
196 array_size -= deletions_count;
197 Array = new type_array[array_size];
198 copy(Array, temp_Array, array_size);
199 delete[] temp_Array;
200}

204 const type_array *old_array,

210 {

212 }

380template <typename type_array>
381void ALGOR::ARRAYDATA<type_array>::resize(const asize_t &NEW_SIZE,
382 const type_array &setElement)
383{
384 if (NEW_SIZE == this->ARRAY->array_size || NEW_SIZE == 0)
385 {
386 throw EXCEPTION_SET::size_failure();
387 }
388 Array<type_array> *OLD_ARRAY = this->ARRAY, *NEW_ARRAY = create_struct<type_array>(NEW_SIZE);
389 asize_t min_size = minimum(OLD_ARRAY->array_size, NEW_ARRAY->array_size);
390 ArrayProcessing<type_array>::copy(NEW_ARRAY->array, OLD_ARRAY->array, min_size);
391 for (asize_t i = OLD_ARRAY->array_size; i < NEW_ARRAY->array_size; i++)
392 {
393 NEW_ARRAY->array[i] = setElement;
394 }
395 this->ARRAY = NEW_ARRAY;
396 remove_struct<type_array>(OLD_ARRAY);
397}

412{
413 asize_t left_limit = 0, right_limit = this->ARRAY->array_size - 1;
414 for (asize_t i = 0; i < this->ARRAY->array_size / 2; i++)
415 {

417 left_limit++;
418 right_limit--;

430template <typename type_array>
431Array<asize_t> *ALGOR::ARRAYDATA<type_array>::searcherOccurrencesOfSubstring(Array<type_array> *&SUBARRAY,
432 ArrayType ArrType)
433{
434 Array<asize_t> *Occurrences = new Array<asize_t>;
435 for (asize_t i = 0; i <= this->ARRAY->array_size - SUBARRAY->array_size; i++)
436 {
437 for (asize_t j = 0; j < SUBARRAY->array_size; j++)
438 {
439 if (this->ARRAY->array[i + j] == SUBARRAY->array[j])
440 {
441 if ((ARRAYDATA::ArrayType)(SUBARRAY->array_size - j) == ArrType)
442 {
443 ArrayProcessing<asize_t>::addElement(
444 Occurrences->array,
445 Occurrences->array_size,
446 i,
447 Occurrences->array_size
448 );
449 }
450 }
451 else
452 {
453 break;
454 }
455 }
456 }
457 if (Occurrences->array_size == 0)
458 {
459 throw EXCEPTION_SET::not_found();
460 }
461 return Occurrences;
462}

464template <typename type_array>
465type_array ALGOR::ARRAYDATA<type_array>::average()
466{
467 type_array average = 0;
468 for (asize_t i = 0; i < this->ARRAY->array_size; i++)
469 {
470 average += this->ARRAY->array[i];
471 }
472 if (this->ARRAY->array_size == 0)
473 {
474 throw EXCEPTION_SET::division_by_zero("Any non-empty array can get into the class, or the class itself can create it."
475 "However, you can get the array structure from the class and access the data"
476 "directly. The author has only one explanation: you somehow reset the size of"
477 "the array through the structure and now the class works with broken data - in"
478 "the average() method, the division by the size of the array takes place, which"
479 "became equal to zero...");
480 }
481 return average / this->ARRAY->array_size;
482}

494template <typename type_array>
495typename ALGOR::ARRAYDATA<type_array>::mode *ALGOR::ARRAYDATA<type_array>::moda()
496{
497 //Виділення пам'яті під данні
498 mode *modes = new mode;
499 modes->array_size = 1;
500 modes->array = new type_array[1];
501 ubit32_t current_frequency = 0;
502
503 //Пошук першого елементу з максимальною частотою
504 for (asize_t i = 0; i < this->ARRAY->array_size; i++)
505 {
506 current_frequency++;
507 if (i == this->ARRAY->array_size - 1 || this->ARRAY->array[i] != this->ARRAY->array[i + 1])
508 {
509 if (modes->highest_frequency < current_frequency)
510 {
511 modes->highest_frequency = current_frequency;
512 modes->array[0] = this->ARRAY->array[i];
513 }
514 current_frequency = 0;
515 }
516 }
517
518 //Якщо всі значення унікальні - кидається виключення
519 if (modes->highest_frequency == 1)
520 {
521 delete[] modes->array;
522 delete (modes);
523 throw EXCEPTION_SET::value_failure("And to be precise: there is missing value with the maximum frequency. All"
524 "elements in the array are unique and repeated once, so to save resources and time,"
525 "an exception is thrown so that the method does not create a copy of the array"
526 "further");
527 }
528
529 //Пошук всіх наступних елементів з цією частотою
530 current_frequency = 0;
531 for (asize_t i = 0; i < this->ARRAY->array_size; i++)
532 {
533 if (modes->array[0] == this->ARRAY->array[i])
534 {
535 for (asize_t j = i + modes->highest_frequency; j < this->ARRAY->array_size; j++)
536 {
537 current_frequency++;
538 if (j == this->ARRAY->array_size - 1 || this->ARRAY->array[j] != this->ARRAY->array[j + 1])
539 {
540 if (current_frequency == modes->highest_frequency)
541 {
542 ArrayProcessing<type_array>::addElement(
543 modes->array,
544 modes->array_size,
545 this->ARRAY->array[j],
546 modes->array_size
547 );
548 }
549 current_frequency = 0;
550 }
551 }
552 break;
553 }
554 }
555 return modes;
556}

577 {

579 {
